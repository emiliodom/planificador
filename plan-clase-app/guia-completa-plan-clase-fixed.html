function PDFExport({ planData }) {
  const exportToPDF = async () => {
    const element = document.getElementById('plan-content');
    
    try {
      const canvas = await html2canvas(element, {
        scale: 2,
        useCORS: true,
        allowTaint: true,
        backgroundColor: '#ffffff'
      });

      const imgData = canvas.toDataURL('image/png');
      const pdf = new jsPDF('p', 'mm', 'a4');

          <h2 className="text-lg font-bold mb-2">Recursos</h2>
          <p>{planData.recursos}</p>
        </div>
      </div>
      
      <button
        onClick={exportToPDF}
        className="mt-4 bg-red-600 text-white px-6 py-3 rounded-lg hover:bg-red-700"
      >
        <i className="fas fa-download mr-2"></i>
        Exportar a PDF
      </button>
    </div>
  );
}
